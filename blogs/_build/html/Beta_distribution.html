@interact(a=(0, 10, 0.01), b=(0, 10, 0.01))
def update(a=2, b=2):
    with fig.batch_update():
        x = np.linspace(beta.ppf(0.01, a, b),
                beta.ppf(0.99, a, b), 100)
        scatt.x = x
        scatt.y = beta.pdf(x, a, b)
